function getComponentInfo(obj) {
  var results = [ ];

  for (var i = 0; i < obj.components.length; i++) {
    var c = obj.components[i]; 
    if (!c.exports || c.exports.length !== 1) continue;
    var m = c.exports[0].match(/(.*)\/(\w*) (.*)/);
    if (!m) continue;

    var cInfo = {
      moduleName: m[1],
      componentName: m[2],
      moduleVersion: m[3]
    };

    for (var enumIndex = 0; enumIndex < c.enums.length; enumIndex++) {
      var enumInfo = c.enums[enumIndex];
      for (var enumValue in enumInfo.values) {
        //results.push( { moduleName: cInfo.moduleName, componentName: cInfo.componentName, type: 'enum', name: enumValue } );
        results.push("enum %1.%2.%3".arg(cInfo.moduleName).arg(cInfo.componentName).arg(enumValue));
      }
    }
    for (var methodIndex = 0; methodIndex < c.methods.length; methodIndex++) {
      var methodInfo = c.methods[methodIndex];
      results.push("method %1.%2.%3".arg(cInfo.moduleName).arg(cInfo.componentName).arg(methodInfo.name));
    }
    for (var propertyIndex = 0; propertyIndex < c.properties.length; propertyIndex++) {
      var propertyInfo = c.properties[propertyIndex];
      results.push("property %1.%2.%3".arg(cInfo.moduleName).arg(cInfo.componentName).arg(propertyInfo.name));
    }
    for (var signalIndex = 0; signalIndex < c.signals.length; signalIndex++) {
      var signalInfo = c.signals[signalIndex];
      results.push("signals %1.%2.%3".arg(cInfo.moduleName).arg(cInfo.componentName).arg(signalInfo.name));
    }
  }
  return results;
}
